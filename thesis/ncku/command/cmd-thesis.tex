
% Some helper function about thesis

% ----------------------------------------------------------------------------

\newcommand{\BeginThesis}[0]
{
  % Start of paper
  \begin{document}

  % Set the line spacing for whole thesis
  \baselineskip = 26pt
} % End of \newcommand{}

\newcommand{\EndThesis}[0]
{
  % End of paper
  \end{document}
} % End of \newcommand{}
