\makeatletter

% Regular text spacing
\def\doublespacing{%
    \def\default@spacing{\baselineskip=20pt plus .5pt minus .2pt}}
\def\onehalfspacing{%
    \def\default@spacing{\baselineskip=20.5pt plus .5pt minus .2pt}}
\def\singlespacing{%
    \def\default@spacing{\baselineskip=15.5pt plus .5pt minus .2pt}}
\def\specialspacing{%
    \def\default@spacing{\baselineskip=21.5pt plus .5pt minus .2pt}}

\makeatother

% Default lines are double spaced

\newcommand{\VarDefaultLinesSpacingTypeEng}{0}
\newcommand{\VarDefaultLinesSpacingTypeChi}{1}
\newcommand{\VarDefaultLinesSpacingType}{%
  \VarDefaultLinesSpacingTypeEng} % Default
\newcommand{\GetDefaultLinesSpacingType}{%
  \VarDefaultLinesSpacingType}
\newcommand{\SetDefaultLinesSpacingType}[1]{\renewcommand{\VarDefaultLinesSpacingType}{#1}}

\newcommand{\UseEngLinesSpacing}
{%
  \doublespacing%
  \doublespace%
} % End of \newcommand{}

\newcommand{\UseChiLinesSpacing}
{%
  \singlespacing%
  \singlespace%
} % End of \newcommand{}

\newcommand{\ThesisWroteInChi}{%
  \SetDefaultLinesSpacingType{\VarDefaultLinesSpacingTypeChi}}

\newcommand{\UseDefaultLinesSpacing}
{%
  \ifthenelse{\GetDefaultLinesSpacingType = \VarDefaultLinesSpacingTypeEng}%
  {\UseEngLinesSpacing}{}%
  %
  \ifthenelse{\GetDefaultLinesSpacingType = \VarDefaultLinesSpacingTypeChi}%
  {\UseChiLinesSpacing}{}%
} % End of \newcommand{}

% ----------------------------------------------------------------------------

% Default lines space is following English style format 
\UseEngLinesSpacing
